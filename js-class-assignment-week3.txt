<script>
	function generate(){
	
		var letters = "0123456789ABCDEF";
	
		var color = '#';
	
		for (var i = 0; i < 6; i++)
			color += letters[(Math.floor(Math.random() * 16))];
	
		var b=document.getElementById("body");
		var p=document.getElementById("code");
		p.innerText=color;

		b.style.backgroundColor=`${color}`;
	}
</script>
